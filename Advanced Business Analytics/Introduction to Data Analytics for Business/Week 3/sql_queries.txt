1. 

SELECT COUNT(*)
FROM PLANES

2. 

SELECT TAIL_NUMBER
FROM PLANES
WHERE SEAT_COUNT >= 100
ORDER BY SEAT_COUNT ASC

3.

SELECT COUNT(FLIGHT_NUMBER) AS SFN, TAIL_NUMBER
FROM FLIGHTS
GROUP BY TAIL_NUMBER
ORDER BY SFN DESC

4.

SELECT COUNT(*)
FROM FLIGHTS
GROUP BY TAIL_NUMBER
HAVING SUM(PASSENGER_COUNT) > 600

5.

SELECT COUNT(FLIGHT_NUMBER), b.COUNTRY as COUNTRY
FROM FLIGHTS a
LEFT JOIN AIRPORTS b
ON a.DEPARTURE_AIRPORT = b.AIRPORT
GROUP BY b.COUNTRY

6.

SELECT COUNT(FLIGHT_NUMBER), b.REGIONALITY as REGIONALITY
FROM FLIGHTS a
LEFT JOIN CITY_PAIRS b
ON a.DEPARTURE_AIRPORT = b.DEPARTURE_AIRPORT
GROUP BY b.REGIONALITY
ORDER BY COUNT(FLIGHT_NUMBER) DESC

7.

SELECT COUNT(*)
FROM CITY_PAIRS
WHERE DEPARTURE_AIRPORT IN ('KLAX', 'KDEN', 'KORD', 'KDET', 'KLGA')

8.

SELECT COUNT(*)
FROM AIRPORTS 
WHERE ELEVATION IS NULL

9. 

SELECT PASSENGER_COUNT, FLIGHT_NUMBER
FROM FLIGHTS 
ORDER BY PASSENGER_COUNT ASC
